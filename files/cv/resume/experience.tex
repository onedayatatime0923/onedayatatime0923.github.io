\cventry
{DesCyPhy Lab, Prof. Pierluigi Nuzzo}
{Berkeley, CA}
{Graduate Student Researcher}
{01/2021 -- Present}
{
\vspace{-3.5mm}
\item {\descriptiontitlestyle{Neural Controller Transformation for Formal Verification}}
\begin{cvitems}
\item Developed a \textbf{provably exact transformation} from ReLU neural-network controllers to soft decision-tree representations, making solver-based verification more scalable and interpretable.
\item Proved correctness and demonstrated closed-loop safety certification on CartPole, MountainCar, and CarRacing benchmarks; resulted in a \textbf{CDC 2023 publication}.
\end{cvitems}
\vspace{1.8mm}
\item {\descriptiontitlestyle{Robustness Contracts for Learning-Enabled CPS}}
\begin{cvitems}
\item Introduced \textbf{contract-based robustness specifications} that decouple plant dynamics from neural controllers for compositional safety verification.
\item Designed an \textbf{SMT-based refinement algorithm} that incrementally tightens layer-wise abstraction bounds to accelerate safe-reachability analysis under bounded uncertainty.
\end{cvitems}
\vspace{1.8mm}
\item {\descriptiontitlestyle{Assume–Guarantee Contract Optimization}}
\begin{cvitems}
\item Formulated \textbf{MILP-based contract-refinement} techniques to automatically synthesize tighter assume-guarantee specifications.
\item Improved realizability and reduced verification complexity in multi-agent autonomous-driving scenarios.
\end{cvitems}
\vspace{1.8mm}
\item {\descriptiontitlestyle{Deep RL Controller Distillation for Verifiability}}
\begin{cvitems}
\item Developed a controller-distillation framework that extracts compact, interpretable control laws from deep RL policies.
\item Applied optimization-based refinement to preserve task performance while improving closed-loop stability and formal verifiability; resulted in an \textbf{Allerton 2022 publication}.
\end{cvitems}
\vspace{-3.5mm}
}
